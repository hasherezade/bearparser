187QString FileHdrWrapper::translateFieldContent(size_t fieldId)
188{
189 IMAGE_FILE_HEADER * hdr = reinterpret_cast<IMAGE_FILE_HEADER*>(getPtr());
190 if (!hdr) return "";
191
